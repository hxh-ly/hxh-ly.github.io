# 0. 目的

理解什么是 Hook Event？明白其使用的场景 深入理解 Hook Event 的实现原理

# 1. 什么是 HookEvent

HookEvent———— 是自定义事件结合生命周期实现的一种从外部为组件注入额外生命周期方法的功能

# 2.使用场景

//第三方组件
mounted(){
   var res= await axios.get('xxx')
   arr=res

}

假设我要在 mounted 请求数据的时候添加 loading，而由于是第三方组件不好修改，我们就可以通过hookEvent为生命周期注入额外方法

# 3.实现原理

我们知道 vue 在各种生命周期阶段内部是调用callHook的执行钩子方法的，

export function callHook(vm: component, hook: string) {
  //打开依赖收集
  pushTarget();
  const handler = vm.options[hook];
  const info = `${hook} hook`;
  if (handler) {
    handler.forEach((item) => {
      //包裹了执行方法 做些try catch
      invokeWithErrorHanding(item, vm, null, vm, info);
    });
  }
  //如果设置event hook  <compA  hook:mounted='xxx'>
  if (vm._hasHookEvent) {
    vm.emit("hook:" + hook);
  }
  //关闭收集依赖
}

export invokeWithErrorHanding(
  handler:Function,
  context:any,
  args:null|| any[],
  vm:any,
  info:string
  } 
  {
    let res
    try{
      res= args? handler.apply(vm,args):handler.call(vm)
      if(res&&  isPromise(res)&& res._handled ) {
        res.catch(err=>{
          handleError(e,vm,info+'(promise/async)')
        })
        res._handled=true
      }
    }catch(err) {
        handleError(e,vm,info)
    }
  }
)

Vue.prototype.$on=function(name:string||any[],fn:Function):Component {
   const vm = this
  if(Array.isArray(name)) {
    name.forEach(item=>{
      vm.$on(item,fn)
    })
  }else {
    (vm._events[name||vm.events[name]=[]).push(fn)
    //如果监听的事件存在hookEvent 标记 _hasHookVent
